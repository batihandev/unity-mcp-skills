# uitk_create_from_template

Generate a paired UXML + USS file pair from a named starter template.

**Signature:** `UitkCreateFromTemplate(template string, savePath string, name string = null)`

**Returns:** `{ success, template, ussPath, uxmlPath, name }`

**Notes:**
- `template` must be one of: `menu`, `hud`, `dialog`, `settings`, `inventory`, `list`, `tab-view`, `toolbar`, `card`, `notification`.
- `savePath` is the directory where both files will be written.
- Fails if either file already exists at the destination.

**Prerequisites:** [`execution_result`](../_shared/execution_result.md), [`validate`](../_shared/validate.md), [`workflow_manager`](../_shared/workflow_manager.md)

```csharp
using UnityEngine;
using UnityEditor;
using System.IO;
using System.Globalization;

internal class CommandScript : IRunCommand
{
    public void Execute(ExecutionResult result)
    {
        string template = "menu";
        string savePath = "Assets/UI/Generated";
        string name = null;

        if (Validate.Required(template, "template") is object tErr) { result.SetResult(tErr); return; }
        if (Validate.SafePath(savePath, "savePath") is object pErr) { result.SetResult(pErr); return; }

        var dir = savePath.TrimEnd('/', '\\');
        var uiName = !string.IsNullOrEmpty(name)
            ? name
            : CultureInfo.InvariantCulture.TextInfo.ToTitleCase(template.ToLower());

        var ussFilePath  = $"{dir}/{uiName}.uss";
        var uxmlFilePath = $"{dir}/{uiName}.uxml";

        if (File.Exists(ussFilePath) || File.Exists(uxmlFilePath))
        {
            result.SetResult(new { error = $"Files already exist at {dir}/{uiName}.[uss|uxml]" });
            return;
        }

        if (!Directory.Exists(dir))
            Directory.CreateDirectory(dir);

        // Template content is generated by the upstream skill engine
        // based on the template name — use the actual skill call in production.
        // This skeleton demonstrates the file-creation flow.
        string ussContent  = $"/* {uiName} — generated from template: {template} */\n";
        string uxmlContent = $"<?xml version=\"1.0\" encoding=\"utf-8\"?>\n<engine:UXML xmlns:engine=\"UnityEngine.UIElements\">\n    <Style src=\"{uiName}.uss\" />\n</engine:UXML>\n";

        File.WriteAllText(ussFilePath,  ussContent,  System.Text.Encoding.UTF8);
        File.WriteAllText(uxmlFilePath, uxmlContent, System.Text.Encoding.UTF8);
        AssetDatabase.ImportAsset(ussFilePath);
        AssetDatabase.ImportAsset(uxmlFilePath);

        var ussAsset  = AssetDatabase.LoadAssetAtPath<UnityEngine.Object>(ussFilePath);
        var uxmlAsset = AssetDatabase.LoadAssetAtPath<UnityEngine.Object>(uxmlFilePath);
        if (ussAsset  != null) WorkflowManager.SnapshotObject(ussAsset,  SnapshotType.Created);
        if (uxmlAsset != null) WorkflowManager.SnapshotObject(uxmlAsset, SnapshotType.Created);

        result.SetResult(new { success = true, template, ussPath = ussFilePath, uxmlPath = uxmlFilePath, name = uiName });
    }
}
```
